## 5. 详细技术方案

### 5.1 架构设计

#### 5.1.1 数据库扩展

```prisma
// Video 模型扩展
model Game {
  // ... 现有字段
  videos Video[]
}

model Video {
  id String @id @default(cuid())
  video_id String @unique // YouTube/Twitch video ID
  platform VideoPlatform
  game_id String?
  game Game? @relation(fields: [game_id], references: [id])

  title String
  description String?
  thumbnail_url String
  channel_name String
  channel_url String
  duration Int? // 秒
  view_count BigInt?
  published_at DateTime?

  video_type VideoType
  video_tags Json @default("[]")

  is_live Boolean @default(false)
  is_featured Boolean @default(false)

  // AI元数据
  ai_summary String?
  ai_relevance_score Float?

  created_at DateTime @default(now())
  updated_at DateTime @updatedAt

  @@index([game_id])
  @@index([platform])
  @@index([published_at])
}

enum VideoPlatform {
  youtube
  twitch
  tiktok
  youtube_shorts
}

enum VideoType {
  gameplay
  tutorial
  review
  walkthrough
  news
  trailer
  live
}
```
